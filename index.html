function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); if (!ss) { throw new Error("Apps Script nije vezan za Google Sheet."); } const sessionsSheet = getOrCreateSheet_(ss, "WorkoutSessions", [ "session_id", "date", "body_weight", "duration_min", "total_volume", "density", "exercise_count", "created_at" ]); const exerciseLogSheet = getOrCreateSheet_(ss, "ExerciseLog", [ "session_id", "date", "exercise_name", "muscle", "sets", "reps", "weight", "volume", "e1rm", "performed", "created_at" ]); const payload = JSON.parse(e.postData.contents || "{}"); const sessionId = String(payload.id || Date.now()); const date = payload.date || ""; const bodyWeight = toNumber_(payload.body_weight); const durationMin = toNumber_(payload.duration_min); const totalVolume = toNumber_(payload.total_volume); const density = toNumber_(payload.density); const exercises = Array.isArray(payload.exercises) ? payload.exercises : []; const createdAt = new Date(); const performedCount = exercises.filter(ex => toNumber_(ex.sets) > 0 && toNumber_(ex.reps) > 0 && toNumber_(ex.weight) > 0 ).length; sessionsSheet.appendRow([ sessionId, date, bodyWeight, durationMin, totalVolume, density, performedCount, createdAt ]); const exerciseRows = exercises.map(ex => { const sets = toNumber_(ex.sets); const reps = toNumber_(ex.reps); const weight = toNumber_(ex.weight); const volume = sets * reps * weight; const e1rm = calculateE1RM_(weight, reps); const performed = sets > 0 && reps > 0 && weight > 0 ? 1 : 0; return [ sessionId, date, ex.name || "", ex.muscle || "", sets, reps, weight, volume, e1rm, performed, createdAt ]; }); if (exerciseRows.length > 0) { const startRow = exerciseLogSheet.getLastRow() + 1; exerciseLogSheet.getRange(startRow, 1, exerciseRows.length, exerciseRows[0].length).setValues(exerciseRows); } formatSheets_(sessionsSheet, exerciseLogSheet); return jsonResponse_({ status: "success", message: "Workout saved.", session_id: sessionId }); } catch (error) { return jsonResponse_({ status: "error", message: error.message || "Unknown error" }); } } function doGet() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); if (!ss) { throw new Error("Apps Script nije vezan za Google Sheet."); } const sessionsSheet = ss.getSheetByName("WorkoutSessions"); const exerciseLogSheet = ss.getSheetByName("ExerciseLog"); if (!sessionsSheet || !exerciseLogSheet || sessionsSheet.getLastRow() < 2 || exerciseLogSheet.getLastRow() < 2) { return jsonResponse_({ status: "success", last_session: null }); } const lastSessionRow = sessionsSheet.getLastRow(); const lastSessionId = String(sessionsSheet.getRange(lastSessionRow, 1).getValue()); const lastSessionDate = sessionsSheet.getRange(lastSessionRow, 2).getValue(); const exerciseData = exerciseLogSheet.getDataRange().getValues(); const headers = exerciseData[0]; const rows = exerciseData.slice(1); const sessionIdIndex = headers.indexOf("session_id"); const nameIndex = headers.indexOf("exercise_name"); const muscleIndex = headers.indexOf("muscle"); const setsIndex = headers.indexOf("sets"); const repsIndex = headers.indexOf("reps"); const weightIndex = headers.indexOf("weight"); const lastSessionExercises = rows .filter(r => String(r[sessionIdIndex]) === lastSessionId) .map(r => ({ name: r[nameIndex], muscle: r[muscleIndex], sets: r[setsIndex], reps: r[repsIndex], weight: r[weightIndex] })); return jsonResponse_({ status: "success", last_session: { id: lastSessionId, date: lastSessionDate, exercises: lastSessionExercises } }); } catch (error) { return jsonResponse_({ status: "error", message: error.message || "Unknown error" }); } } function getOrCreateSheet_(ss, sheetName, headers) { let sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); } const lastRow = sheet.getLastRow(); const lastCol = sheet.getLastColumn(); if (lastRow === 0) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); } else { const existingHeaders = sheet.getRange(1, 1, 1, Math.max(lastCol, headers.length)).getValues()[0]; const normalizedExisting = existingHeaders.slice(0, headers.length).map(String); const normalizedTarget = headers.map(String); if (JSON.stringify(normalizedExisting) !== JSON.stringify(normalizedTarget)) { sheet.clear(); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); } } return sheet; } function formatSheets_(sessionsSheet, exerciseLogSheet) { sessionsSheet.setFrozenRows(1); exerciseLogSheet.setFrozenRows(1); sessionsSheet.autoResizeColumns(1, sessionsSheet.getLastColumn()); exerciseLogSheet.autoResizeColumns(1, exerciseLogSheet.getLastColumn()); } function calculateE1RM_(weight, reps) { const w = toNumber_(weight); const r = toNumber_(reps); if (w <= 0 || r <= 0 || r >= 37) { return 0; } const denominator = 1.0278 - 0.0278 * r; if (denominator <= 0) { return 0; } return Math.round((w / denominator) * 10) / 10; } function toNumber_(value) { if (typeof value === "number") { return Number.isFinite(value) ? value : 0; } if (typeof value === "string") { const normalized = value.replace(",", ".").trim(); if (normalized === "") return 0; const parsed = Number(normalized); return Number.isFinite(parsed) ? parsed : 0; } return 0; } function jsonResponse_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
